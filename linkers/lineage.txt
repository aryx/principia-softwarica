#--------------------------------------------------------------------------
# Binary formats
#--------------------------------------------------------------------------

*a.out
  1971
  # Ken Thompson, original UNIX binary format
  # name from default output of 'as', produced by ld

*COFF
  1983
  a.out
  # AT&T, Common Object File Format, UNIX System V
  # added section headers and symbolic debug info

*ELF
  1992
  COFF
  # AT&T/USL, for System V Release 4
  # standard on Linux/BSD/Solaris, produced by GNU ld/Gold/lld/mold

*Mach-O
  1988
  a.out
  # CMU Mach project, used by NeXTStep then macOS/iOS
  # produced by Apple ld64

*PE
  1993
  COFF
  # Microsoft, Portable Executable, derived from COFF
  # produced by Microsoft Linker for Windows NT
